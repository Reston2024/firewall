```bash
#!/bin/bash
# validate-phase10.sh — Phase 10 validation suite: Telemetry Migration to Malcolm
# Run FROM dev machine (Windows Git Bash / WSL). SSHes to supportTAK-server.
# Usage: bash scripts/validate-phase10.sh [--full]
# Quick mode: MAL-02, MAL-03, MIG-02, MIG-03 smoke checks
# Full mode (--full): includes live Suricata trigger test (MAL-02 integration, ~120s wait)

set -euo pipefail
if [ "${1:-}" = "--full" ]; then FULL=1; else FULL=0; fi

SSH_TARGET="opsadmin@192.168.1.22"
SSH_IPFIRE="root@192.168.1.1"
SSH_IPFIRE_KEY="$HOME/.ssh/ipfire_ed25519"
FAIL=0; PASS=0; SKIP=0

pass() { echo "PASS: $1"; PASS=$((PASS + 1)); }
fail() { echo "FAIL: $1"; FAIL=$((FAIL + 1)); }
skip() { echo "SKIP: $1"; SKIP=$((SKIP + 1)); }

echo "=== Phase 10 Validation Suite — Telemetry Migration to Malcolm — $(date) ==="
echo ""

# Discover Malcolm internal credentials
INTERNAL_CREDS=$(ssh ${SSH_TARGET} \
  "docker exec malcolm-dashboards-helper-1 cat /var/local/curlrc/.opensearch.primary.curlrc \
   | grep '^user:' | sed 's/^user: \"//;s/\"$//'" 2>/dev/null)

if [ -z "$INTERNAL_CREDS" ]; then
  echo "ERROR: Cannot retrieve Malcolm internal credentials. Is Malcolm running?"
  exit 1
fi

# --- MAL-02: Suricata EVE JSON in Malcolm OpenSearch ---
echo "[MAL-02] Suricata EVE JSON events in Malcolm OpenSearch"
MAL02_COUNT=$(ssh ${SSH_TARGET} \
  "docker exec malcolm-opensearch-1 curl -sk -u '${INTERNAL_CREDS}' \
   'https://localhost:9200/_search?q=event.module:suricata&size=0' 2>/dev/null" \
  | python3 -c "import sys,json; d=json.load(sys.stdin); print(d.get('hits',{}).get('total',{}).get('value',0))" 2>/dev/null || echo "0")

if [ "$MAL02_COUNT" -gt 0 ] 2>/dev/null; then
  pass "MAL-02: ${MAL02_COUNT} Suricata events in Malcolm OpenSearch"
else
  fail "MAL-02: No Suricata events in Malcolm OpenSearch. Check Filebeat: ssh ${SSH_TARGET} 'sudo journalctl -u filebeat --since \"5 min ago\" --no-pager | tail -10'"
fi
echo ""

# --- MAL-02-full: Live Suricata alert trigger (--full only) ---
if [ "$FULL" -eq 1 ]; then
  echo "[MAL-02-full] Triggering live Suricata alert via testmynids.org"
  ssh -i "$SSH_IPFIRE_KEY" -o StrictHostKeyChecking=no -o ConnectTimeout=10 ${SSH_IPFIRE} \
    'curl -s http://testmynids.org/uid/index.html > /dev/null' 2>/dev/null || true
  echo "  Waiting 120 seconds for pipeline: Suricata -> SCP cron -> Filebeat -> Logstash -> OpenSearch..."
  sleep 120
  MAL02F_COUNT=$(ssh ${SSH_TARGET} \
    "docker exec malcolm-opensearch-1 curl -sk -u '${INTERNAL_CREDS}' \
     'https://localhost:9200/_search?q=event.module:suricata&size=0' 2>/dev/null" \
    | python3 -c "import sys,json; d=json.load(sys.stdin); print(d.get('hits',{}).get('total',{}).get('value',0))" 2>/dev/null || echo "0")
  if [ "$MAL02F_COUNT" -gt "$MAL02_COUNT" ] 2>/dev/null; then
    pass "MAL-02-full: New Suricata alert appeared (${MAL02_COUNT} -> ${MAL02F_COUNT})"
  else
    fail "MAL-02-full: No new alert after trigger (still ${MAL02F_COUNT}). Check SCP cron and Filebeat logs."
  fi
  echo ""
fi

# --- MAL-03: IPFire syslog in Malcolm OpenSearch ---
echo "[MAL-03] IPFire syslog entries (FORWARDFW) in Malcolm OpenSearch"
MAL03_COUNT=$(ssh ${SSH_TARGET} \
  "docker exec malcolm-opensearch-1 curl -sk -u '${INTERNAL_CREDS}' \
   'https://localhost:9200/_search?q=FORWARDFW&size=0' 2>/dev/null" \
  | python3 -c "import sys,json; d=json.load(sys.stdin); print(d.get('hits',{}).get('total',{}).get('value',0))" 2>/dev/null || echo "0")

if [ "$MAL03_COUNT" -gt 0 ] 2>/dev/null; then
  pass "MAL-03: ${MAL03_COUNT} syslog entries with FORWARDFW in Malcolm OpenSearch"
else
  fail "MAL-03: No FORWARDFW syslog entries in Malcolm. Check rsyslog relay: ssh ${SSH_TARGET} 'cat /etc/rsyslog.d/20-malcolm-forward.conf'"
fi
echo ""

# --- MIG-01: Parallel validation (informational — Loki may be decommissioned) ---
echo "[MIG-01] Parallel validation status"
LOKI_STATUS=$(ssh ${SSH_TARGET} "docker ps --format '{{.Names}}' 2>/dev/null | grep -c loki" || echo "0")
if [ "$LOKI_STATUS" -gt 0 ]; then
  pass "MIG-01: Loki still running — parallel validation window active"
else
  pass "MIG-01: Loki decommissioned — parallel window completed (MIG-02 applied)"
fi
echo ""

# --- MIG-02: No Loki stack containers ---
echo "[MIG-02] Loki/Alloy/Grafana/Prometheus containers removed"
OLD_CONTAINERS=$(ssh ${SSH_TARGET} "docker ps --format '{{.Names}}' 2>/dev/null | grep -iE 'loki|alloy|grafana|prometheus|node-exporter' || echo ''")

if [ -z "$OLD_CONTAINERS" ]; then
  pass "MIG-02: No Loki stack containers in docker ps"
else
  fail "MIG-02: Old containers still running: ${OLD_CONTAINERS}"
fi
echo ""

# --- MIG-03: Malcolm containers healthy ---
echo "[MIG-03] Malcolm containers healthy"
MALCOLM_HEALTHY=$(ssh ${SSH_TARGET} "docker compose -f /opt/malcolm/docker-compose.yml ps --format '{{.Name}} {{.Status}}' 2>/dev/null | grep -c 'Up'" || echo "0")
if [ "$MALCOLM_HEALTHY" -ge 20 ]; then
  pass "MIG-03: ${MALCOLM_HEALTHY} Malcolm containers running"
else
  fail "MIG-03: Only ${MALCOLM_HEALTHY} Malcolm containers Up (expected 20+)"
fi
echo ""

# --- MIG-03b: Filebeat system service active ---
echo "[MIG-03b] Filebeat system service active"
FB_STATUS=$(ssh ${SSH_TARGET} "systemctl is-active filebeat 2>/dev/null" || echo "inactive")
if [ "$FB_STATUS" = "active" ]; then
  pass "MIG-03b: Filebeat system service is active"
else
  fail "MIG-03b: Filebeat service is ${FB_STATUS}. Run: ssh ${SSH_TARGET} 'sudo systemctl start filebeat'"
fi
echo ""

# --- MIG-03c: rsyslog relay forwarding to Malcolm :5514 ---
echo "[MIG-03c] rsyslog relay forwarding to Malcolm :5514"
RSYSLOG_FWD=$(ssh ${SSH_TARGET} "cat /etc/rsyslog.d/20-malcolm-forward.conf 2>/dev/null | grep '5514'" || echo "")
if [ -n "$RSYSLOG_FWD" ]; then
  pass "MIG-03c: rsyslog relay config contains port 5514"
else
  fail "MIG-03c: /etc/rsyslog.d/20-malcolm-forward.conf missing or does not contain 5514"
fi
echo ""

# --- MIG-04: Runbook updated (grep for deprecated terms) ---
echo "[MIG-04] Telemetry runbook — no deprecated references"
if [ -f "docs/telemetry-deployment-runbook.md" ]; then
  DEPRECATED=$(grep -ciE 'alloy|loki\.source|loki\.write|prometheus\.scrape|grafana.*3000|3100/ready' docs/telemetry-deployment-runbook.md 2>/dev/null || echo "0")
  HAS_FILEBEAT=$(grep -ci 'filebeat' docs/telemetry-deployment-runbook.md 2>/dev/null || echo "0")
  HAS_MALCOLM=$(grep -ci 'malcolm' docs/telemetry-deployment-runbook.md 2>/dev/null || echo "0")
  if [ "$DEPRECATED" -eq 0 ] && [ "$HAS_FILEBEAT" -gt 0 ] && [ "$HAS_MALCOLM" -gt 0 ]; then
    pass "MIG-04: Runbook references Filebeat (${HAS_FILEBEAT}x) and Malcolm (${HAS_MALCOLM}x), no deprecated terms"
  elif [ "$DEPRECATED" -gt 0 ]; then
    fail "MIG-04: Runbook still contains ${DEPRECATED} deprecated references (Alloy/Loki/Prometheus/Grafana:3000)"
  else
    fail "MIG-04: Runbook missing Filebeat or Malcolm references"
  fi
else
  fail "MIG-04: docs/telemetry-deployment-runbook.md not found"
fi
echo ""

# --- IPFire syslog target check ---
echo "[INFO] IPFire syslog.conf target IP"
IPFIRE_SYSLOG=$(ssh -i "$SSH_IPFIRE_KEY" -o StrictHostKeyChecking=no -o ConnectTimeout=10 ${SSH_IPFIRE} \
  'grep "192.168" /etc/syslog.conf 2>/dev/null' 2>/dev/null || echo "UNREACHABLE")
if echo "$IPFIRE_SYSLOG" | grep -q "192.168.1.22"; then
  pass "IPFire syslog.conf points to 192.168.1.22 (correct)"
elif echo "$IPFIRE_SYSLOG" | grep -q "192.168"; then
  fail "IPFire syslog.conf points to wrong IP: ${IPFIRE_SYSLOG}. Update via WUI > Logs > Log Settings"
else
  skip "IPFire SSH unreachable — verify syslog target manually"
fi
echo ""

# --- Summary ---
echo "=== Phase 10 Validation ==="
echo "=== CHECKS COMPLETE: $PASS PASS, $FAIL FAIL, $SKIP SKIP ==="
if [ $FAIL -gt 0 ]; then
  exit 1
else
  exit 0
fi
```
