<!-- Rollback script template pattern from research -->
```bash
#!/bin/bash
# rollback-{category}.sh — Restore {config} from backup
# Usage: bash rollback-{category}.sh [backup-file]
# If no backup-file given, lists available backups and exits
# Run ON IPFire

ROLLBACK_DIR="/root/rollback"
CONFIG_FILE="{target_config_path}"
CATEGORY="{category}"
BACKUP_FILE="$1"

if [ -z "$BACKUP_FILE" ]; then
  echo "Available $CATEGORY backups:"
  ls -lt "$ROLLBACK_DIR"/${CATEGORY}-*.bak 2>/dev/null || echo "  None found"
  echo ""
  echo "Usage: bash rollback-${CATEGORY}.sh $ROLLBACK_DIR/${CATEGORY}-YYYYMMDD-HHMMSS.bak"
  exit 0
fi
# ... restore and reload
```
